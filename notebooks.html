class Distribution:
    def __init__(self):
        self.n_vals_total = 766294
        self.n_blocks_day = 7200
        self.v_return_block = 0.4
        self.v_return_block_median = 0.1
        self.v_return_none = 0.0025
        self.v_val = 32
        
    def calculate(self, n_days: int, n_vals: int, cum = False):
        self.n_vals = n_vals
        def _return_for_day(x: bool):
            if x:
                return self.v_return_block
            return self.v_return_none

        self.win_prob = self.n_blocks_day / self.n_vals_total  # probability of winning
        self.rtrn_prob = (self.win_prob * _return_for_day(True) + (1-self.win_prob) * _return_for_day(False)) * self.n_vals

        # Initialize array to store returns for each iteration
        days = list(range(1, n_days+1, 1))
        returns = []

        for day in days:
            rtrns = [_return_for_day(item) for item in (np.random.random(n_vals) < self.win_prob)]  # each ticket either wins or loses
            returns.append(np.sum(rtrns))  # total return for the day

        # Calculate standard deviation
        self.std_returns = np.std(returns)
        self.rlzd_means = np.mean(returns)

        print(f"Covering {n_vals} validators -- {n_vals*self.v_val} ETH - ${round(n_vals*1.6/1000,2)}m")

        self.df = pd.DataFrame()
        self.df['Days'] = days
        self.df['Returns'] = returns
        self.df['Returns_cumulative'] = np.cumsum(returns)
    
    def plot_returns(self, cumulative = False):
        y_data = 'Returns_cumulative' if cumulative else 'Returns'
        
        if cumulative:
            return_e = self.rtrn_prob * max(self.df.Days)
            return_r = np.sum(self.df.Returns)
            daily = ''
        else:
            return_e = self.rtrn_prob
            return_r = np.mean(self.df.Returns)
            daily = ' daily'
        print(f"Expected{daily} return: {round(return_e, 3)} ETH")
        print(f"Realized{daily} return: {round(return_r, 3)} ETH")
        print(f"Shortfall: {round(return_r / return_e * 100 - 100, 3)}%")
        
        # Create a bar graph of the returns
        fig = px.bar(self.df, x='Days', y=y_data,
            height=500, title = f"Daily return {'cumulative' if cumulative else ''}")
        fig.update_layout(hovermode="x", yaxis= {"tickformat": '.4f' })

        # Add lines for 1 standard deviation in each direction
        max_x = max(self.df.Days)
        fig.add_trace(go.Scatter(x=[1, max_x], 
                                 y=[self.rtrn_prob, self.rtrn_prob * (max_x if cumulative else 1)], mode='lines', name='Mean'))
        fig.add_trace(go.Scatter(x=[1, max_x], 
                                 y=[self.rtrn_prob - self.std_returns, self.rtrn_prob * (max_x if cumulative else 1) - self.std_returns], mode='lines', name='Mean - 1 SD'))
        fig.add_trace(go.Scatter(x=[1, max_x], 
                                 y=[self.rtrn_prob + self.std_returns, self.rtrn_prob * (max_x if cumulative else 1) + self.std_returns], mode='lines', name='Mean + 1 SD'))

        fig.show()

lose_amount = win_passive / n_vals_total  # Passive return per block

# Calculate the probability of winning in a single chance
prob_win_single_chance = n_blocks_per_day / n_vals_total

# Initialize an array to store the returns for each day
daily_returns = np.zeros(n_days)

# Simulate the lottery for each day
np.random.seed(42)  # For reproducibility
for day in range(n_days):
    # Simulate the outcome for each validator for this day
    outcomes = np.random.rand(n_vals) < prob_win_single_chance
    # Calculate the total returns for this day
    daily_returns[day] = np.sum(outcomes) * win_amount + (n_vals - np.sum(outcomes)) * lose_amount

variance_daily_return = n_vals * (
    prob_win_single_chance * (win_amount)**2 + 
    (1 - prob_win_single_chance) * (lose_amount)**2
) - prob_win_single_chance**2 # https://math.oxford.emory.edu/site/math117/expectedValueVarianceOfBinomial/

# Insurance guarantee
guaranteed_return = expected_total_return_all_days
mean_expected_return = expected_total_return_all_days

# Calculate the Z-score for the guaranteed return
Z_score = (guaranteed_return - mean_expected_return) / std_total_return_all_days

# Calculate the probability of getting a return less than the guaranteed return
probability_of_payout = norm.cdf(Z_score)

# Calculate the payout amount (difference between guaranteed amount and lower CI limit)
payout_amount = mean_expected_return - lower_bound

# Calculate the expected payout
expected_payout = probability_of_payout * payout_amount

# Assume a profit margin of 20%
profit_margin = 0.2

# Calculate the premium for the insurance
premium = expected_payout * (1 + profit_margin)
